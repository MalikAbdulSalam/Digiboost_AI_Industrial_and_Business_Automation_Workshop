Complete Guide to YOLO Object Detection, Security Systems, and Industrial Automation

1. YOLO (You Only Look Once)

YOLO is a state-of-the-art, real-time object detection system. Unlike traditional classifiers that scan images with sliding windows, YOLO applies a single neural network to the full image, dividing it into regions and predicting bounding boxes and probabilities for each region.

рҹ”‘ Key Advantages:
  • Extremely fast (45-155 frames per second)
  • Sees the entire image context (global reasoning)
  • Learns generalizable object representations
  • Excellent for real-time applications like security, autonomous vehicles, and industrial automation

2. Ultralytics

Ultralytics is the official implementation of YOLO, providing a simple yet powerful Python interface for training, validation, prediction, and export of YOLO models.

d) Detect Person and Trigger Alarm

import cv2
from ultralytics import YOLO
import winsound
import time

# CAMERA CONFIG
ip = "192.168.1.4022"
url = f"rtsp://admin:123456@{ip}:554/unicast/c1/s0/live"

# ALARM SETTINGS
last_alarm_time = 0
alarm_delay = 3

def trigger_alarm():
    global last_alarm_time
    now = time.time()
    if now - last_alarm_time > alarm_delay:
        print("рҹҡЁ PERSON DETECTED!")
        winsound.Beep(1000, 500)
        last_alarm_time = now

    if result.boxes is not None:
        for box in result.boxes:
            cls_id = int(box.cls[0])
            if cls_id == 0:  # person class

    if person_found:
        trigger_alarm()

    annotated_frame = result.plot()
    cv2.imshow("AI CCTV Security System", annotated_frame)

    if cv2.waitKey(1) == 27:

e) Detect Person in Specific Region (ROI) and Trigger Alarm

import cv2
import subprocess
import numpy as np
from ultralytics import YOLO
import winsound
import time

# CAMERA CONFIG
ip = "192.16128.1.42223"
url = f"rtsp://admin:123456@{ip}:554/unicast/c1/s0/live"

ffmpeg_path = r"C:\ffmpeg\bin\ffmpeg.exe"  # Update your path

# YOLO MODEL
model = YOLO("yolo11n.pt")

# ALARM SETTINGS
last_alarm_time = 0
alarm_delay = 3

def trigger_alarm():
    global last_alarm_time
    now = time.time()
    if now - last_alarm_time > alarm_delay:
        print("рҹҡЁ PERSON ENTERED RESTRICTED ZONE!")
        winsound.Beep(1000, 500)
        last_alarm_time = now

# RESTRICTED ZONE (ROI POLYGON)
ROI = np.array([
    (400, 200), (1600, 200),
    (1600, 900), (400, 900)
], np.int32)

# FFmpeg PIPE
command = [ffmpeg_path, "-rtsp_transport", "tcp", "-fflags", "nobuffer",
           "-flags", "low_delay", "-i", url, "-f", "rawvideo",
           "-pix_fmt", "bgr24", "-"]

pipe = subprocess.Popen(command, stdout=subprocess.PIPE, stderr=subprocess.DEVNULL)

width, height = 1920, 1080
frame_size = width * height * 3

while True:
    raw_frame = pipe.stdout.read(frame_size)
    if len(raw_frame) != frame_size:
        continue

    frame = np.frombuffer(raw_frame, dtype=np.uint8).reshape((height, width, 3))
    frame = frame.copy()

    cv2.polylines(frame, [ROI], True, (0, 0, 255), 2)

    results = model(frame, conf=0.4, verbose=False)
    result = results[0]
    person_in_zone = False

    if result.boxes is not None:
        for box in result.boxes:
            cls_id = int(box.cls[0])
            if cls_id == 0:
                x1, y1, x2, y2 = map(int, box.xyxy[0])
                cv2.rectangle(frame, (x1, y1), (x2, y2), (0, 255, 0), 2)

                person_center = ((x1 + x2)//2, (y1 + y2)//2)
                inside = cv2.pointPolygonTest(ROI, person_center, False)

                if inside >= 0:
                    person_in_zone = True
                    cv2.circle(frame, person_center, 5, (0, 0, 255), -1)

    if person_in_zone:
        trigger_alarm()

    cv2.imshow("Restricted Zone AI Security", frame)

    if cv2.waitKey(1) == 27:
        break

pipe.terminate()
cv2.destroyAllWindows()

4. Pretrained Weights

Pretrained models are models that someone else has trained on large datasets. We can use these for transfer learning or direct inference.

рҹ“Ұ Where to find pretrained models:

  • Roboflow Universe - Thousands of public models and datasets
  • Hugging Face Hub - Extensive model repository
  • Ultralytics - Official pretrained weights (yolo11n.pt, yolo11s.pt, etc.)

Training Command:

yolo task=detect mode=train model=yolo11n.pt data=data.yaml epochs=100 imgsz=640
вҡҷпёҸ Parameter Explanation:
  • task=detect - Object detection task
  • mode=train - Training mode
  • model=yolo11n.pt - Pretrained weights (nano version)
  • data=data.yaml - Dataset configuration file
  • epochs=100 - Number of training iterations
  • imgsz=640 - Input image size

7. Practical Project: Detect Wrong Bottle on Conveyor Belt

Problem Statement: On a production line, we need to ensure only correct products move forward. If a Coca-Cola bottle appears on the Pepsi conveyor belt (or vice versa), we must detect it and trigger an alarm/removal mechanism.

"""
Conveyor Belt Quality Control System
Detects wrong bottles on specific conveyor belts and triggers removal
"""

from ultralytics import YOLO
import cv2
import numpy as np
import time
from playsound import playsound

# CONFIGURATION
COCA_COLA_CLASS_ID = 0
PEPSI_CLASS_ID = 1

# Define conveyor belt regions (ROI)
CONVEYOR_BELTS = {
    "belt_1": {
        "roi": np.array([(50, 300), (600, 300), (600, 700), (50, 700)], np.int32),
        "allowed_bottle": "Pepsi",
        "disallowed_bottle": "CocaCola"
    },
    "belt_2": {
        "roi": np.array([(650, 300), (1200, 300), (1200, 700), (650, 700)], np.int32),
        "allowed_bottle": "CocaCola",
        "disallowed_bottle": "Pepsi"
    }
}

# Alarm settings
last_alarm_time = 0
alarm_cooldown = 3

# LOAD MODEL
model = YOLO("Trained_Models/bottle_detector.pt")

# VIDEO SOURCE
cap = cv2.VideoCapture("conveyor_belt.mp4")

def trigger_alarm(bottle_type, belt_name):
    global last_alarm_time
    current_time = time.time()
    
    if current_time - last_alarm_time > alarm_cooldown:
        print(f"рҹҡЁ ALERT: {bottle_type} detected on {belt_name}!")
        print(f"   вҶ’ Activating removal mechanism...")
        try:
            playsound("wrong_bottle_alarm.mp3")
        except:
            pass  # Fallback to system beep
        last_alarm_time = current_time

# MAIN DETECTION LOOP
while True:
    ret, frame = cap.read()
    if not ret:
        break
    
    frame = frame.copy()
    
    # DRAW CONVEYOR BELT ZONES
    for belt_name, belt_info in CONVEYOR_BELTS.items():
        roi = belt_info["roi"]
        allowed = belt_info["allowed_bottle"]
        cv2.polylines(frame, [roi], True, (0, 255, 0), 2)
        cv2.putText(frame, f"{belt_name} (Only {allowed})", 
                    (roi[0][0], roi[0][1] - 10),
                    cv2.FONT_HERSHEY_SIMPLEX, 0.6, (0, 255, 0), 2)
    
    # YOLO INFERENCE
    results = model(frame, conf=0.5, verbose=False)
    result = results[0]
    
    # PROCESS DETECTIONS
    if result.boxes is not None:
        for box in result.boxes:
            cls_id = int(box.cls[0])
            confidence = float(box.conf[0])
            x1, y1, x2, y2 = map(int, box.xyxy[0])
            class_name = model.names[cls_id]
            
            center_x = (x1 + x2) // 2
            center_y = (y1 + y2) // 2
            
            # Determine which belt
            bottle_belt = None
            for belt_name, belt_info in CONVEYOR_BELTS.items():
                roi = belt_info["roi"]
                if cv2.pointPolygonTest(roi, (center_x, center_y), False) >= 0:
                    bottle_belt = belt_name
                    break
            
            color = (0, 255, 0)
            
            if bottle_belt:
                belt_info = CONVEYOR_BELTS[bottle_belt]
                if class_name == belt_info["disallowed_bottle"]:
                    color = (0, 0, 255)
                    trigger_alarm(class_name, bottle_belt)
                    cv2.putText(frame, f"WRONG! Remove {class_name}",
                                (x1, y1 - 20), cv2.FONT_HERSHEY_SIMPLEX,
                                0.7, (0, 0, 255), 2)
                elif class_name == belt_info["allowed_bottle"]:
                    cv2.putText(frame, f"вң“ {class_name} OK",
                                (x1, y1 - 20), cv2.FONT_HERSHEY_SIMPLEX,
                                0.6, (0, 255, 0), 2)
            
            cv2.rectangle(frame, (x1, y1), (x2, y2), color, 2)
            label = f"{class_name} ({confidence:.2f})"
            cv2.putText(frame, label, (x1, y1 - 5),
                        cv2.FONT_HERSHEY_SIMPLEX, 0.5, color, 2)
    
    cv2.imshow("Bottle Quality Control", frame)
    
    if cv2.waitKey(1) == 27:
        break

cap.release()
cv2.destroyAllWindows()
print("System shutdown complete.")
